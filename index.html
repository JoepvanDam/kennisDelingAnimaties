CSS @keyframes

Je kan een CSS @keyframes animatie maken door met procenten aan te geven welke stijl van een element verandert.

Je kan ook From en To gebruiken, maar dit werkt hetzelfde als 0% en 100%. Zie hier een voorbeeld.

#animatedElement {
     animation: example1Animation 5s ease-in-out 1s infinite alternate none running;
}

@keyframes example1Animation {
    From { top: 0; }
    To { top: calc(100% - 50px); }
}

@keyframes example1Animation {
    0% { top: 0; }
    100% { top: calc(100% - 50px); }
}

CSS selector & transition

Met een CSS selector als :hover, :focus, etc., en een defined transition voor dat element,
kan je een simpele animatie maken. Zie hier een aantal voorbeelden.

#animatedHover { width: 50px; transition: all 1000ms; }
 #animatedHover:hover { width: 200px; }
#animatedFocus { background-color: white; transition: all 1s; }
 #animatedFocus:focus { background-color: blue; }
#animatedChecked { outline: 5px solid white; transition: all 1s; }
 #animatedChecked:checked { outline: 5px solid blue; }
#animatedActive {
    background-color: white;
    color: black;
    transition: background-color 500ms, color 5s;
}
 #animatedActive:active {
    background-color: blue;
    color: red;
}

JS styling & CSS transition

Deze methode is nagenoeg hetzelfde als de vorige, maar in plaats van een CSS selector gebruik je JavaScript.

Het grootste verschil is dat je met JavaScript meer vrijheid hebt in wanneer je een animatie start en eindigt.

En je kan ook makkelijker 1 object meerdere bewegingen laten doen. Zie hier weer een voorbeeld.

#animatedElement { transition: all 500ms; }

JS frames

Dit is de meest ingewikkelde, maar ook de leukste methode om animaties mee te maken (vind ik dan).

Je hebt namelijk heel veel ('oneindige') vrijheid. Zie hier een voorbeeld.

function animateElement1() {
    let frame = 0;
    const animatedElement = document.getElementById("animatedElement");

    const id = setInterval(() => {
        if (frame <= 90) {
            animatedElement.style.top = frame+"%";
            frame++;
        }
        else { clearInterval (id); }
    }, 10);
}

function animateElement2() {
    let frame = 0;
    const animatedElement = document.getElementById("animatedElement");

    const id = setInterval(() => {
        if (frame <= 80) {
            animatedElement.style.left = frame+"%";
            frame++;
        }
        else { clearInterval (id); }
    }, 10);
}

function animateElement3() {
    let frame = 0;
    const animatedElement = document.getElementById("animatedElement");

    const id = setInterval(() => {
        if (frame <= 90) {
            animatedElement.style.top = frame+"%";
            animatedElement.style.left = (frame*(8/9))+"%";
            frame++;
        }
        else { clearInterval (id); }
    }, 10);
}

function animateElement4() {
    let frame = 0;
    const animatedElement = document.getElementById("animatedElement");

    const id = setInterval(() => {
        if (frame <= 90) {
            animatedElement.style.top = frame+"%";
            frame++;
        }
        else if (frame <= 170) {
            animatedElement.style.left = (frame-90)+"%";
            frame++;
        }
        else if (frame <= 260) {
            animatedElement.style.left = ((frame*-1)+260)+"%";
            frame++;
        }
        else if (frame <= 340) {
            animatedElement.style.left = ((frame*-1)+340)+"%";
            frame++;
        }
        else { clearInterval (id); }
    }, 10);
}

function animateElement4() {
    let frame = 0;
    let topPos = 0;
    let leftPos = 0;
    const animatedElement = document.getElementById("animatedElement");

    const id = setInterval(() => {
        if (frame <= 80) {
            animatedElement.style.left = leftPos+"%";
            frame++;
            leftPos++;
        }
        else if (frame <= 170) {
            animatedElement.style.top = topPos+"%";
            frame++;
            topPos++;
        }
        else if (frame <= 250) {
            animatedElement.style.left = leftPos+"%";
            frame++;
            leftPos--;
        }
        else if (frame <= 340) {
            animatedElement.style.top = topPos+"%";
            frame++;
            topPos--;
        }
        else { clearInterval (id); }
    }, 10);
}

function animateElementReset() {
    const animatedElement = document.getElementById("animatedElement");

    animatedElement.style.top = "0";
    animatedElement.style.left = "0";
}

#animatedElement {
    height: 10%;
    width: 20%;
}
